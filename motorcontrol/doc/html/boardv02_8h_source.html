144 #define PIN_MODE_OUTPUT(n) (1U << ((n) * 2U))

150 #define PIN_OTYPE_OPENDRAIN(n) (1U << (n))
151 #define PIN_OSPEED_VERYLOW(n) (0U << ((n) * 2U))
152 #define PIN_OSPEED_LOW(n) (1U << ((n) * 2U))
153 #define PIN_OSPEED_MEDIUM(n) (2U << ((n) * 2U))
154 #define PIN_OSPEED_HIGH(n) (3U << ((n) * 2U))
